import math

def find_roots(a=1,b=1,c=1):
    """
    This is a docstring. It is what the help() function returns
    about this function.
    """
    # YOUR CODE HERE
    x = b**2-4*a*c
    
    r1 = (-b - math.sqrt(x))/(2*a)
    r2 = (-b + math.sqrt(x)) /(2*a)
    
    return r1, r2
    
    raise NotImplementedError()
